WITH current AS (
  SELECT id, num_of,
         COALESCE(qte_recue_pf, 0) AS q_init,
         COALESCE(qte_cartonnee,  0) AS q_cartonnee
  FROM of_tracking
  WHERE id = {{ Number(appsmith.store.id) }}
),
updated AS (
  UPDATE of_tracking o
  SET qte_recue_pf = LEAST(
      (SELECT q_init FROM current) + GREATEST({{ Number(Number_qty.value) || 0 }}, 0),
      (SELECT q_cartonnee FROM current)
  )
  FROM current c
  WHERE o.id = c.id
  RETURNING o.id, o.num_of, o.qte_recue_pf AS q_total
)
